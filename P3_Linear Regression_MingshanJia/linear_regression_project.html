        for j in range(col):
            M[i][j] = round(M[i][j], decPts)

def matxMultiply(A, B):
    if len(A[0]) != len(B):
        raise ValueError
    B_T = transpose(B)
    Res = []   
    for row_A in A:
        Row = []
        for col_B in B_T:
            ele = 0
            for a, b in zip(row_A, col_B):
                ele += a*b
            Row.append(ele)
        Res.append(Row)
    return Res

""" Gaussian Jordan  Ax = b.
    
        A:

        epsilon: 0 1.0e-16
        
     x  Ax = b 
    None Ab 
    None A 
"""
def gj_Solve(A, b, decPts=4, epsilon=1.0e-16):
    #Step1: A,bNone
    if len(A) != len(b):
        return None
    
    #Step2: Ab
    Ab = augmentMatrix(A, b)
    
    #Step3: Ab
    for c in range(len(Ab[0])-1):
        row_of_max_ele = 0;       # row
        max_ele = 0;              # 
        for r in range(len(Ab)):
            if r >= c:   #  >=
                if abs(Ab[r][c]) > max_ele:
                    max_ele = abs(Ab[r][c])
                    row_of_max_ele = r
        if max_ele < epsilon:     # 0
            return None
        else:
            if row_of_max_ele != c:  # 
                swapRows(Ab, row_of_max_ele, c)
            scaleRow(Ab, c, 1.0 / Ab[c][c])
            
            # c0
            for rx in range(len(Ab)):
                if rx != c:    # 
                    if abs(Ab[rx][c]) > epsilon:
                        addScaledRow(Ab, rx, c, -Ab[rx][c])
    
    #Step4: Ab
    Res = []
    j = len(Ab)
    for i in range(len(Ab)):
        row = []
        row.append(Ab[i][j])            
        Res.append(row)
    matxRound(Res, decPts = 4)
    
    return Res
